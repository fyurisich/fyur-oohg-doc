///////////////////////////////////////////////////////////////////////////////
// HBPRINTER FONT COMMANDS
///////////////////////////////////////////////////////////////////////////////

DEFINE FONT <cfont>
      [ NAME <cface> ]
      [ SIZE <size> ]
      [ WIDTH <width> ]
      [ ANGLE <angle> ]
      [ BOLD ]
      [ ITALIC ]
      [ UNDERLINE ]
      [ STRIKEOUT ]
Defines a font.

CHANGE FONT <cfont>
      [ NAME <cface> ]
      [ SIZE <size> ]
      [ WIDTH <width> ]
      [ ANGLE <angle> ]
      [ BOLD ]
      [ NOBOLD ]
      [ ITALIC ]
      [ NOITALIC ]
      [ UNDERLINE ]
      [ NOUNDERLINE ]
      [ STRIKEOUT ]
      [ NOSTRIKEOUT ]
Changes attributes in an already defined font.

SELECT FONT <cfont>
Sets the default font.

SET CHARSET <charset>
Sets the default charset.
