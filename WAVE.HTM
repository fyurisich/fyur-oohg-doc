///////////////////////////////////////////////////////////////////////////////
// WAVE COMMANDS
////////////////////////////////////////////////////////////////////////////// 

#command PLAY WAVE <wave> ;
      [<r: FROM RESOURCE>] ;
      [<s: SYNC>];
      [<ns: NOSTOP>] ;
      [<l: LOOP>] ;
      [<nd: NODEFAULT>] ;
   => ;
      PlayWave( <wave>, <.r.>, <.s.>, <.ns.>, <.l.>, <.nd.> )

#command STOP WAVE ;
   => ;
      StopWave()

#command STOP WAVE NODEFAULT ;
   => ;
      StopWave()
